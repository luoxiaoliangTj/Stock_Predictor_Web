<script>
// 搜索函数 - 调用后端API
async function searchStock() {
    const query = document.getElementById('searchInput').value.trim();
    if (!query) return;
    
    const resultsDiv = document.getElementById('results');
    resultsDiv.innerHTML = '<p style="text-align:center; color:#757575; padding:20px;">搜索中...</p>';
    
    try {
        const response = await fetch(`/api/stock/search?q=${encodeURIComponent(query)}`);
        if (!response.ok) {
            throw new Error(`API错误: ${response.status}`);
        }
        
        const results = await response.json();
        
        if (results.length === 0) {
            resultsDiv.innerHTML = '<p style="text-align:center; color:#757575; padding:20px;">未找到匹配的股票</p>';
            return;
        }
        
        resultsDiv.innerHTML = results.slice(0, 10).map(r => 
            `<div class="result-item" onclick="window.location.href='generated/${r.code}.html'">
                <span>${r.name}</span>
                <span class="result-code">${r.code}</span>
            </div>`
        ).join('');
    } catch (error) {
        resultsDiv.innerHTML = `<p style="text-align:center; color:#d32f2f; padding:20px;">搜索失败: ${error.message}</p>`;
    }
}
</script>
